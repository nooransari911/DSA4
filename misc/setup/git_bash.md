## SSH Key generation template
Non-root ::
ssh-keygen -t ed25519 -C "email@address.com"
#This generates two keys: public (non-.pub) and private (*.pub);
mkdir ~/.ssh
mv public_key ~/.ssh
mv private_key.pub ~/.ssh
exec ssh-agent bash
ssh-add ~/.ssh/public_key

Root ::
su
cp -r /home/ansarimn/.ssh/ /.ssh/
exec ssh-agent bash
ssh-add /.ssh/public_key
